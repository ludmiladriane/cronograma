    <script>
        const APP_KEY = "mpes_v2_mobile_storage";
        
        let state = {
            topics: [
                { id: 1, category: "Português", title: "Interpretação de Textos", done: false },
                { id: 2, category: "Português", title: "Acentuação e Ortografia", done: false },
                { id: 3, category: "Informática", title: "Windows 10/11", done: false },
                { id: 4, category: "Informática", title: "Word e Excel", done: false },
                { id: 5, category: "Raciocínio Lógico", title: "Proposições e Conectivos", done: false },
                { id: 6, category: "Direito Adm.", title: "Princípios da Adm. Pública", done: false }
            ],
            logs: [],
            revisions: []
        };

        let timerActive = false;
        let seconds = 0;
        let timerInterval = null;
        let myChart = null;

        function init() {
            const saved = localStorage.getItem(APP_KEY);
            if (saved) state = JSON.parse(saved);
            
            lucide.createIcons();
            showTab('dashboard');
            refreshUI();
            
            // Solicitar permissão para notificações (opcional)
            if ("Notification" in window) Notification.requestPermission();
        }

        function save() {
            localStorage.setItem(APP_KEY, JSON.stringify(state));
        }

        function showTab(tabId) {
            document.querySelectorAll('.tab-content').forEach(el => el.classList.add('hidden'));
            document.getElementById(`tab-${tabId}`).classList.remove('hidden');

            // Atualizar botões (Desktop e Mobile)
            document.querySelectorAll('nav button').forEach(btn => btn.classList.remove('active-tab', 'text-blue-600'));
            
            const btnDt = document.getElementById(`btn-${tabId}-dt`);
            const btnMb = document.getElementById(`btn-${tabId}-mb`);
            
            if(btnDt) btnDt.classList.add('active-tab');
            if(btnMb) {
                btnMb.classList.add('text-blue-600');
                btnMb.classList.remove('text-slate-400');
            }

            if(tabId === 'dashboard') setTimeout(initChart, 100);
        }

        function toggleTimer() {
            const hud = document.getElementById('timer-hud-mb');
            const badge = document.getElementById('mobile-timer-badge');
            const iconMb = document.getElementById('play-icon-mb');

            if (!timerActive) {
                timerActive = true;
                timerInterval = setInterval(() => {
                    seconds++;
                    const timeStr = new Date(seconds * 1000).toISOString().substr(11, 8);
                    document.getElementById('timer-display-dt').innerText = timeStr;
                    document.getElementById('timer-display-mb').innerText = timeStr;
                }, 1000);

                hud.classList.remove('opacity-0', 'translate-y-[-100px]');
                badge.classList.remove('hidden');
                iconMb.setAttribute('data-lucide', 'square');
                document.getElementById('timer-btn-dt').innerText = "Encerrar Sessão";
                document.getElementById('timer-btn-dt').classList.replace('bg-blue-600', 'bg-red-600');
            } else {
                if (seconds > 60) {
                    state.logs.push({ date: new Date().toISOString().split('T')[0], mins: Math.floor(seconds / 60) });
                    save();
                }
                clearInterval(timerInterval);
                timerActive = false;
                seconds = 0;
                hud.classList.add('opacity-0', 'translate-y-[-100px]');
                badge.classList.add('hidden');
                iconMb.setAttribute('data-lucide', 'play');
                document.getElementById('timer-btn-dt').innerText = "Iniciar Foco";
                document.getElementById('timer-btn-dt').classList.replace('bg-red-600', 'bg-blue-600');
                document.getElementById('timer-display-dt').innerText = "00:00:00";
                refreshUI();
            }
            lucide.createIcons();
        }

        function toggleTopic(id) {
            const topic = state.topics.find(t => t.id === id);
            topic.done = !topic.done;
            
            if (topic.done) {
                // Gerar revisões 1, 7, 15, 30
                [1, 7, 15, 30].forEach(days => {
                    const d = new Date();
                    d.setDate(d.getDate() + days);
                    state.revisions.push({
                        topicId: id,
                        title: topic.title,
                        date: d.toISOString().split('T')[0],
                        done: false
                    });
                });
            }
            save();
            refreshUI();
        }

        function completeRevision(index) {
            state.revisions[index].done = true;
            save();
            refreshUI();
        }

        function refreshUI() {
            renderSubjects();
            renderRevisions();
            updateStats();
            lucide.createIcons();
        }

        function updateStats() {
            const totalMins = state.logs.reduce((acc, curr) => acc + curr.mins, 0);
            const doneTopics = state.topics.filter(t => t.done).length;
            const percent = Math.round((doneTopics / state.topics.length) * 100);
            
            const today = new Date().toISOString().split('T')[0];
            const revCount = state.revisions.filter(r => r.date <= today && !r.done).length;

            document.getElementById('stat-hours').innerText = (totalMins / 60).toFixed(1) + 'h';
            document.getElementById('stat-percent').innerText = percent + '%';
            document.getElementById('stat-rev').innerText = revCount;
            
            const badge = document.getElementById('badge-rev');
            revCount > 0 ? badge.classList.remove('hidden') : badge.classList.add('hidden');
        }

        function renderSubjects() {
            const container = document.getElementById('subjects-container');
            const categories = [...new Set(state.topics.map(t => t.category))];

            container.innerHTML = categories.map(cat => `
                <div class="bg-white rounded-2xl border border-slate-200 overflow-hidden">
                    <div class="bg-slate-50 px-4 py-2 border-b border-slate-100 flex justify-between">
                        <span class="text-[10px] font-black uppercase text-slate-400 tracking-widest">${cat}</span>
                    </div>
                    <div class="divide-y divide-slate-100">
                        ${state.topics.filter(t => t.category === cat).map(t => `
                            <div onclick="toggleTopic(${t.id})" class="flex items-center justify-between p-4 active:bg-slate-50 transition-colors">
                                <span class="text-sm font-semibold ${t.done ? 'text-slate-300 line-through' : 'text-slate-700'}">${t.title}</span>
                                <div class="w-6 h-6 rounded-full border-2 ${t.done ? 'bg-green-500 border-green-500' : 'border-slate-200'} flex items-center justify-center">
                                    ${t.done ? '<i data-lucide="check" class="text-white" size="14"></i>' : ''}
                                </div>
                            </div>
                        `).join('')}
                    </div>
                </div>
            `).join('');
        }

        function renderRevisions() {
            const container = document.getElementById('today-list');
            const today = new Date().toISOString().split('T')[0];
            const todayTasks = state.revisions.filter(r => r.date <= today && !r.done);

            if (todayTasks.length === 0) {
                container.innerHTML = `
                    <div class="bg-white p-12 rounded-[40px] text-center border-2 border-dashed border-slate-200">
                        <div class="w-16 h-16 bg-slate-50 rounded-full flex items-center justify-center mx-auto mb-4">
                            <i data-lucide="coffee" class="text-slate-300"></i>
                        </div>
                        <p class="font-bold text-slate-400">Nenhuma revisão para agora.</p>
                    </div>
                `;
                return;
            }

            container.innerHTML = todayTasks.map((t, idx) => `
                <div onclick="completeRevision(${state.revisions.indexOf(t)})" class="bg-white p-4 rounded-2xl border-l-4 border-l-red-500 border border-slate-200 flex items-center justify-between shadow-sm active:scale-95 transition-all">
                    <div>
                        <h4 class="font-bold text-slate-800">${t.title}</h4>
                        <p class="text-[10px] text-red-500 font-bold uppercase">Pendente</p>
                    </div>
                    <div class="w-10 h-10 rounded-xl bg-slate-50 flex items-center justify-center text-slate-300">
                        <i data-lucide="circle"></i>
                    </div>
                </div>
            `).join('');
        }

        function initChart() {
            const ctx = document.getElementById('chartWeekly').getContext('2d');
            const last7Days = [...Array(7)].map((_, i) => {
                const d = new Date();
                d.setDate(d.getDate() - i);
                return d.toISOString().split('T')[0];
            }).reverse();

            const chartData = last7Days.map(date => {
                const dayLogs = state.logs.filter(l => l.date === date);
                return dayLogs.reduce((acc, curr) => acc + (curr.mins / 60), 0).toFixed(1);
            });

            if (myChart) myChart.destroy();
            myChart = new Chart(ctx, {
                type: 'bar',
                data: {
                    labels: last7Days.map(d => d.split('-')[2]),
                    datasets: [{
                        label: 'Horas',
                        data: chartData,
                        backgroundColor: '#3b82f6',
                        borderRadius: 6
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: { legend: { display: false } },
                    scales: {
                        y: { beginAtZero: true, grid: { display: false } },
                        x: { grid: { display: false } }
                    }
                }
            });
        }

        function exportData() {
            const dataStr = "data:text/json;charset=utf-8," + encodeURIComponent(JSON.stringify(state));
            const downloadAnchorNode = document.createElement('a');
            downloadAnchorNode.setAttribute("href", dataStr);
            downloadAnchorNode.setAttribute("download", "estudos_backup.json");
            document.body.appendChild(downloadAnchorNode);
            downloadAnchorNode.click();
            downloadAnchorNode.remove();
        }

        function importData(event) {
            const reader = new FileReader();
            reader.onload = function(e) {
                state = JSON.parse(e.target.result);
                save();
                window.location.reload();
            };
            reader.readAsText(event.target.files[0]);
        }

        init();
    </script>
